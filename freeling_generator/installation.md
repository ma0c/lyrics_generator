# Installing Freeling from source for Ubuntu

```bash
sudo apt install \
    libboost-dev \
    libboost-regex-dev \
    libicu-dev \
    libboost-system-dev \
    libboost-program-options-dev \
    libboost-thread-dev \
    libboost-filesystem-dev \
    zlib1g
    
wget \
  -O freeling.zip \
  https://github.com/TALP-UPC/FreeLing/releases/download/4.2/FreeLing-src-4.2.zip
  
unzip freeling.zip
cd FreeLing-4.2
```
